## 📚 使用示例

### 示例 1：创建 React 组件

**需求**：
```
创建一个 React 组件，显示用户卡片。
卡片应该包含：
- 用户头像
- 用户名
- 邮箱
- 操作按钮（编辑、删除）
使用 Tailwind CSS 美化。
```

**Codex 生成的代码**：

```javascript
import React from 'react';

export default function UserCard({ user, onEdit, onDelete }) {
  return (
    <div className="bg-white rounded-lg shadow-md p-6 max-w-sm">
      {/* 头像 */}
      <div className="flex justify-center mb-4">
        <img
          src={user.avatar}
          alt={user.name}
          className="w-20 h-20 rounded-full object-cover"
        />
      </div>

      {/* 用户信息 */}
      <div className="text-center mb-4">
        <h2 className="text-xl font-bold text-gray-800">{user.name}</h2>
        <p className="text-gray-600">{user.email}</p>
      </div>

      {/* 操作按钮 */}
      <div className="flex gap-2">
        <button
          onClick={() => onEdit(user.id)}
          className="flex-1 bg-blue-500 hover:bg-blue-600 text-white font-bold py-2 px-4 rounded cursor-pointer"
        >
          编辑
        </button>
        <button
          onClick={() => onDelete(user.id)}
          className="flex-1 bg-red-500 hover:bg-red-600 text-white font-bold py-2 px-4 rounded cursor-pointer"
        >
          删除
        </button>
      </div>
    </div>
  );
}
```

### 示例 2：创建 API 端点

**需求**：
```
创建一个 Express API 端点，获取用户列表。
需要：
- 支持分页（page, limit）
- 支持搜索（search）
- 支持排序（sort）
- 返回 JSON 格式
```

**Codex 生成的代码**：

```javascript
const express = require('express');
const router = express.Router();

// 模拟数据
const users = [
  { id: 1, name: '张三', email: 'zhangsan@example.com' },
  { id: 2, name: '李四', email: 'lisi@example.com' },
  // ... 更多用户
];

// GET /api/users
router.get('/users', (req, res) => {
  let result = [...users];

  // 搜索
  if (req.query.search) {
    result = result.filter(user =>
      user.name.includes(req.query.search) ||
      user.email.includes(req.query.search)
    );
  }

  // 排序
  if (req.query.sort) {
    const [field, order] = req.query.sort.split(':');
    result.sort((a, b) => {
      if (order === 'desc') {
        return b[field] > a[field] ? 1 : -1;
      }
      return a[field] > b[field] ? 1 : -1;
    });
  }

  // 分页
  const page = parseInt(req.query.page) || 1;
  const limit = parseInt(req.query.limit) || 10;
  const start = (page - 1) * limit;
  const end = start + limit;

  const paginatedResult = result.slice(start, end);

  res.json({
    data: paginatedResult,
    total: result.length,
    page,
    limit,
    pages: Math.ceil(result.length / limit)
  });
});

module.exports = router;
```

### 示例 3：修复 Bug

**问题代码**：
```javascript
function calculateDiscount(price, discountPercent) {
  return price - (price * discountPercent);
}
```

**问题**：
- 当 discountPercent 是 0.2（20%）时，应该返回 80% 的价格
- 但实际上返回的是 price - (price * 0.2) = 0.8 * price，这是对的
- 但如果 discountPercent 是 20（表示 20%），就会出错

**Codex 修复**：

```javascript
function calculateDiscount(price, discountPercent) {
  // 确保 discountPercent 是 0-1 之间的小数
  const discount = discountPercent > 1 ? discountPercent / 100 : discountPercent;
  
  // 返回折扣后的价格
  return price * (1 - discount);
}

// 使用示例
console.log(calculateDiscount(100, 0.2));  // 80
console.log(calculateDiscount(100, 20));   // 80
```

### 示例 4：代码审查

**代码**：
```javascript
function getUserData(userId) {
  const user = db.query('SELECT * FROM users WHERE id = ' + userId);
  return user;
}
```

**Codex 审查**：

```
🔴 严重问题：SQL 注入漏洞

问题：
直接拼接用户输入到 SQL 查询，攻击者可以执行任意 SQL 命令。

例子：
userId = "1; DROP TABLE users; --"
查询变成：SELECT * FROM users WHERE id = 1; DROP TABLE users; --

解决方案：
使用参数化查询

修复代码：
function getUserData(userId) {
  const user = db.query('SELECT * FROM users WHERE id = ?', [userId]);
  return user;
}

🟡 中等问题：缺少错误处理

问题：
如果查询失败，函数会崩溃。

解决方案：
添加 try-catch

修复代码：
function getUserData(userId) {
  try {
    const user = db.query('SELECT * FROM users WHERE id = ?', [userId]);
    return user;
  } catch (error) {
    console.error('查询失败:', error);
    return null;
  }
}
```

### 示例 5：性能优化

**问题代码**：
```javascript
function findUser(users, id) {
  for (let i = 0; i < users.length; i++) {
    if (users[i].id === id) {
      return users[i];
    }
  }
  return null;
}
```

**问题**：
- 时间复杂度：O(n)
- 对于大数据集很慢

**Codex 优化**：

```javascript
// 方案 1：使用 Map（最快）
// 时间复杂度：O(1)
const userMap = new Map(users.map(u => [u.id, u]));

function findUser(id) {
  return userMap.get(id) || null;
}

// 方案 2：使用 find（简单）
// 时间复杂度：O(n)，但代码更简洁
function findUser(users, id) {
  return users.find(u => u.id === id) || null;
}

// 方案 3：使用 filter（不推荐）
// 时间复杂度：O(n)，效率低
function findUser(users, id) {
  const result = users.filter(u => u.id === id);
  return result.length > 0 ? result[0] : null;
}

// 性能对比（100,000 个用户）
// Map：< 1ms
// find：~5ms
// filter：~10ms
```
